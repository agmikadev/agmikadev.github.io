    Current App.tsx:
    ```tsx
    import MainFrame from './components/MainFrame';
    import './App.css';

    function App() {
      return (
        <div>
          <MainFrame />
        </div>
      );
    }
    ```

    Replace with:
    ```tsx
    import { useState, useEffect } from 'react';
    import MainFrame from './components/MainFrame';
    import { LoadingSpinner } from './components/LoadingSpinner/LoadingSpinner';
    import './App.css';

    function App() {
      const [isLoading, setIsLoading] = useState(true);

      useEffect(() => {
        // Dismiss spinner after first render cycle completes
        // requestAnimationFrame ensures the 3D scene has at least started rendering
        const timer = requestAnimationFrame(() => {
          requestAnimationFrame(() => {
            setIsLoading(false);
          });
        });
        return () => cancelAnimationFrame(timer);
      }, []);

      return (
        <div>
          {isLoading ? <LoadingSpinner /> : <MainFrame />}
        </div>
      );
    }

    export default App;
    ```

    Key implementation details:
    - Use double `requestAnimationFrame` to ensure the browser has painted at least one frame before dismissing the spinner
    - Import `LoadingSpinner` from the new component path
    - Do NOT add any timeout-based fallback (no setTimeout) — the rAF approach is sufficient for a SPA with no async data fetching
    - The `isLoading` state starts as `true` and transitions to `false` after the first paint cycle
  </action>
